# JS-继承

JS的继承是基于原型链实现的

如果想实现类似JAVA的继承，则需要在子类中进行拷贝父类的属性

# 原型继承

构造器函数有一个prototype属性,会将实例的__proto__指针指向prototype对象，实例由此访问父类的方法

function Father(){}
Father.prototype.message = 'come from Father!'
Father.cars = []

function Son(){}
Son.prototype = new Father() // 重点: 将子类的prototype引用改为Father的实例，而Father的实例又能够访问到Father的prototype属性

const son = new Son()

son.message // come from Father!

# 构造函数

为了解决上述问题，可以更改子类的构造函数实现

function Socket () {
    this.sendMessage = ()=>{  }
}

function SocketSDK () {
    Socket.call(this)
    this.module = {}
}

const webIMApp = new SocketSDK()
const wxIMApp = new SocketSDK()

webIMApp.sendMessage = ()=>2
wxIMApp.sendMessage = ()=>3

console.log(webIMApp.sendMessage(), wxIMApp.sendMessage())
// 2 3

# 寄生式继承

在构造实例的时候，我们想给这个对象添加一个方法，又不想用字面量的形式创建

function extends(proto){
    function Sub(){}

    let o = new proto
    Sub.prototype = o
}

目前，此方法被称作原型式继承，实例的属性不会共享 此方法类似于Object.create(proto)

function Cars () {
    this.wheels = []
}

function extend (proto){
    
    function Sub(){}

    Sub.prototype = new proto

    return new Sub
}

const bwm = Object.create(new Cars) // const bwm = Object.create(new Cars)
const yq = Object.create(new Cars) // const yq = Object.create(new Cars)

bwm.wheels.push('red')

console.log(
    bwm.wheels, // ['red']
    yq.wheels   // []

# 添加实例独有方法

所以我们现在实现了在继承的时候拦截原型链， 为当前实例的原型上添加一个方法

寄生继承

function extend(porto, fnName, fn){
    function Sub() {}

    Sub.prototype = proto

    Sub.prototype.[fnName] = fn // 寄生方法

    return new Sub
}
