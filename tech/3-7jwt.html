  • jwt的认证

我们可以把认证的方法做成一个中间函数，添加到其他路由中


function authenticate(req, res, next) {
    const authorization = req.headers['authorization']
    const token = authorization && authorization.split(' ')[1]

    if (token == null) return res.sendStatus(403)

    jwt.verify(token, process.env.ACCESS_TOKEN_SECRET, (err, user) => {
        if (err) return res.sendStatus(403)
        req.user = user
        next()
    })
}

  • 现在的代码
require('dotenv').config()
const bodyParser = require('body-parser')
const express = require('express')
const jwt = require('jsonwebtoken')

const app = express()
// 解析body
app.use(bodyParser.urlencoded({
    extended: false
}))


const posts = [{  // 模拟数据库 文章列表
    name: 'kyle',
    title: 'first'
}, {
    name: 'kite',
    title: 'second'
}]

app.get('/posts', authenticate, (req, res) => { // 如果认证成功，拿到token中的用户信息
    const user = req.user

    console.log(user)
    res.json(
        posts.filter(post => post.name === user.name)
    )
})

app.post('/login', (req, res) => { 
    // 登录时拿到表单中的username并且根据username生成一个token
    const user = {
        name: req.body.username
    }
    // user的信息加密在token中
    const accessToken = jwt.sign(user, process.env.ACCESS_TOKEN_SECRET)

    res.json({
        accessToken: accessToken
    })
})

function authenticate(req, res, next) {
    const authorization = req.headers['authorization']
    const token = authorization && authorization.split(' ')[1]

    if (token == null) return res.sendStatus(403) // 认证失败 ： 没有token
 
    jwt.verify(token, process.env.ACCESS_TOKEN_SECRET, (err, user) => {
        if (err) return res.sendStatus(403) // 认证失败 ： token不匹配(匹配该用户的特定token)此处在.env文件中只设置了单一的secret
        req.user = user //
        next()
    })
}

app.listen(3000)
