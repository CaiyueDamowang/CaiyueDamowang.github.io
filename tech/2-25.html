const path = require("path");
const webpack = require("webpack");

module.exports = {
  entry: "./src/main.js",
  output: {
    path: path.resolve(__dirname, "dist/"),
    publicPath: "/dist/", // dev 的默认输出路径  会覆盖dev的publicPath 并且是众多loader的参考
    filename: "build.js"
  },
  devServer: {
    // dev是将文件保存在内存中

    // publicPath: 设置之后我们访问的文件，都在这个虚拟路径下，
    // contentBase: 将某一文件作为base路径
    // historyApiCallback: true, // 如果找不到页面则回到默认路径页面
    overlay: true // 在页面查看编译发生的错误 不需要终端
  },
  resolve: {
    alias: {
      // 配置路径的别名
      vue$: "vue/dist/vue.esm.js" // $ 精准匹配 esm(预编译+运行时 版本)
    }
  }
};

这个时候还要注意的一点就是在 webpack.config.js 文件里面

如果配置了 output 的 publicPath 这个字段的值的话，在 index.html 文件里面也应该做出调整。

因为 webpack-dev-server 伺服的文件是相对 publicPath 这个路径的。因此，如果你的 webpack.config.js 配置成这样的：

        publicPath: '/dist/',  // dev 的默认输出路径

# 添加 css （sass）

将webpack中的module改写

module: {
  rules: [
    {
      test: /\.css$/,
      use: ["vue-style-loader", "css-loader"] // loader 顺序 从后往前
      // css-loader使得我们可以用模块化的写法引入css
      // vue-style-loader会将引入的css插入到html页面里的style标签里
    },
    {
      test: /\.sass$/, // 添加sass 匹配规则
      use: ["vue-style-loader", "css-loader", "sass-loader?indentedSyntax"]
    }
  ];
}

# 添加 babel

ES6 的语法大多数浏览器依旧不支持,bable 可以把 ES6 转码成 ES5 语法，这样我们就可以大胆的在项目中使用最新特性了

# babel 速览

babel 总共分为三个阶段：解析，转换，生成。

我们需要知道现在 babel 本身是不具备这种转化功能，提供这些转化功能的是一个个 plugin。

所以我们没有配置任何 plugin 的时候，经过 Babel 输出的代码是没有改变的

在目录中新建.babelrc文件

stage-3会使代码支持aync/await功能

{
    "presets": [
      ["env", { "modules": false }],
      "stage-3"
    ]
}

最后完整的webpack.config.js如下

const path = require('path')
const webpack = require('webpack')
const VueLoaderPlugin = require('vue-loader/lib/plugin')
module.exports = {
    entry: './src/main.js',
    output: {
        path: path.resolve(__dirname, 'dist/'),
        publicPath: '/dist/',  // dev 的默认输出路径
        filename: 'build.js'
    },
    devServer: {
        // historyApiCallback: true, // 如果找不到页面则回到默认路径页面
        overlay: true // 在页面查看编译发生的错误 不需要终端
    },
    module: {
        rules: [
            {
                test: /\.css$/,
                use: ['vue-style-loader', 'css-loader'] // loader 顺序 从后往前
                                                        // css-loader使得我们可以用模块化的写法引入css
                                                        // vue-style-loader会将引入的css插入到html页面里的style标签里
            },
            {
                test: /\.sass$/,                        // 添加sass 匹配规则
                use: ['vue-style-loader', 'css-loader', 'sass-loader?indentedSyntax']
            },
            {
                test: /\.js$/,
                use: ['babel-loader'],
                exclude: '/node_modules/' // 表示不包括此路径的文件
            },
            {
                test: /\.vue$/,
                loader: 'vue-loader',
                options: {
                    loaders: {
                        scss: ['vue-style-loader', 'css-loader', 'sass-loader?indextedSyntax']
                    }
                }
            }
        ]
    },
    plugins: [
        new VueLoaderPlugin()
    ],
    resolve: {
        alias: { // 配置路径的别名
            'vue$': 'vue/dist/vue.esm.js'  // $ 精准匹配 esm(预编译+运行时 版本)
        }
    }
}
