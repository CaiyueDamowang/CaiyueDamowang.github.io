# 在Vue项目中使用D3.js 实现响应式图像

  • D3 负责数据的计算

  • Vue 负责视图管理

很多书籍和教程都是用D3来操作DOM

在Vue项目中，我们把DOM操作交给Vue

# Vue的优点：

方便于我将D3.js的力学模型API单独抽离，计算各个节点的坐标 Vue监听数据改变后，会自动更新视图，不需要hack D3.js的simulation 缺点，Vue会对依赖组件进行Diff

# 生成D3图谱

  • vue负责视图

  • 生成图谱所需要的结点和连线

  • 使用svg的circle 和 line

svg的使用方法： circle 绑定cx、cy 确定圆心的位置 line 绑定 x1、y1 | x2/y2 确定两点连线

    <svg
      :width="width"
      :height="height"
      data-type="svg"
    >
      <circle
        v-for="node in nodes"
        :key="node.id"
        class="svg-circle"
        r="25"
        :cx="node.x"
        :cy="node.y"
        fill="#f38"
        @mousedown="drag"
      />
      <line
        v-for="path in ships"
        :key="path.uuid"
        class="svg-line"
        :x1="path.source.x"
        :y1="path.source.y"
        :x2="path.target.x"
        :y2="path.target.y"
        data-type="line"
        style="stroke:rgb(66,66,66);stroke-width:4"
      />
    </svg>

# 生成视图所需要的数据

data(){

    // ship 传入结点关系 { target: 1, source: 2 }
    // nodes 传入结点原始数据
    // simulation (力学模型)会将节点的坐标进行处理

    function initSimulation(nodes, ships) {
        const simulation = d3.forceSimulation()
            .nodes(nodes)
            .force('link', forceLink(ships).distance(150).id(node => node.id))
            .force('center', forceCenter(this.width / 2, this.height / 2))
            .force('charge', forceManyBody(0.1))
            .force('collision', forceCollide(25))

        return  simulation
    }

    // d3会改变传入的对象 所以直接引用之前的对象
    return {
        nodes: nodes
        ships: ships 
    }
}
