# 堆排序

堆排序是树形选择排序的改进

树形选择排序就像锦标赛一样，父节点为两个子节点的较大值

但是这种排序浪费存储空间较多

堆排序利用元素的上浮和下沉进行了改进

数据结构为线性长度的线性表

# 最小堆

插入： 元素插入在数组最后面，不断上浮比较（直到指针越界，或者符合比较规则）

移除：最小值即为第一个结点， 将数组最后一个结点和第一个结点交换，不断下沉（ 下沉时，左右子节点和此结点进行比较)

class minHeap {
    constructor(data=[]) {
        this.data = []
        this.init(data)
    }
    init(data) {
        for (let key = 0; key < data.length; key++) {
            const element = data[key];
            this.insert(element)
        }
    }
    insert(val) {
        let curPosition =  this.data.length
        this.data[curPosition] = val
        this.minify(curPosition)
    }
    minify(curPosition) {
        // 堆的上浮
        let parent = this._parent(curPosition)
        while(curPosition > 0) {
            if(this.data[parent] < this.data[curPosition]) {
                break
            } else {
                this._swap(curPosition, parent)
                curPosition = parent
                parent = this._parent(curPosition)
            }
        }
    }
    remove() {
        // 堆的下沉
        if(!this.data.length) { return null }
        const min = this.data[0]
        this.data[0] = this.data.pop()
        this.minHeapSort()

        return min
    }
    minHeapSort() {
        let curPosition = 0
        let left = this._left(curPosition)
        let right = this._right(curPosition)
        curPosition = this.data[right]
            && this.data[right] > this.data[left]
            ? left
            : right
        let parent = this._parent(curPosition)
        while(curPosition < this.data.length) {
            console.log(curPosition)
            if(this.data[curPosition] > this.data[parent]) {
                break
            } else {
                this._swap(curPosition, parent)
                left = this._left(curPosition)
                right = this._right(curPosition)

                curPosition = this.data[right]
                    && this.data[left] > this.data[right]
                    ? right
                    : left
                parent = this._parent(curPosition)
            }
        }
    }
    _swap(a, b){
        [
            this.data[a],
            this.data[b]
        ] = [
            this.data[b],
            this.data[a]
        ]
    }
    _parent(i) { return Math.floor( (i - 1)/2) }  
    _right(i)  { return ++i * 2 - 1 }
    _left(i)   { return ++i * 2 }
}
