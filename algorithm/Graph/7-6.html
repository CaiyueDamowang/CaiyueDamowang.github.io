# 被围绕的区域

给定一个二维的矩阵，包含 'X' 和 'O'（字母 O）。

找到所有被 'X' 围绕的区域，并将这些区域里所有的 'O' 用 'X' 填充。

示例：

X X X X
X O O X
X X O X
X O X X
1
2
3
4

输出

X X X X
X X X X
X X X X
X O X X
1
2
3
4

思路： 无权图的遍历

看到模型后想到所有能够连通的O都可以建立连接，由此看出是图模型算法

连接节点则是深度遍历或者广度遍历

两种办法： 一种连接范围内的O (❌) 一种连接边缘O，打上标记，最后清楚标记 (✅)

根据bfs或者dfs来进行递归遍历，顺着边缘的O找到所有连通边缘的路径，并且标记为#

const solve = function (board) {
    if (!board || !board.length) return

    let height = board.length,
        width = board[0].length
    // 先从边缘搜索，标记连通的O为#
    for (let i = 0; i < height; i++) {
        for (let j = 0; j < width; j++) {
            const isEdge = (
                i === 0) ||
                (j === 0) ||
                (i === height - 1) ||
                (j === width - 1)
            if (isEdge && board[i][j] === 'O') {
                dfs(board, i, j)
            }
        }
    }

    for (let i = 0; i < height; i++) {
        for (let j = 0; j < width; j++) {
            if (board[i][j] === 'O') {
                board[i][j] = 'X'
            }
            if (board[i][j] === '#') {
                board[i][j] = 'O'
            }
        }
    }

    function dfs(board, i, j) {
        if (i < 0 || j < 0 || i > height - 1 || j > width - 1) {
            return
        }
        if (board[i][j] === 'O') {
            board[i][j] = '#'
            dfs(board, i + 1, j)
            dfs(board, i - 1, j)
            dfs(board, i, j + 1)
            dfs(board, i, j - 1)
        }
    }
};

const solve = function (board) {
    if (!board || !board.length) return

    const height = board.length,
        width = board[0].length
    const stack = []
    // 先从边缘搜索，标记连通的O为#
    for (let i = 0; i < height; i++) {
        for (let j = 0; j < width; j++) {
            const isEdge = (
                i === 0) ||
                (j === 0) ||
                (i === height - 1) ||
                (j === width - 1)
            if (isEdge && board[i][j] === 'O') {
                dfs(i, j)
            }
        }
    }

    for (let i = 0; i < height; i++) {
        for (let j = 0; j < width; j++) {
            if (board[i][j] === 'O') {
                board[i][j] = 'X'
            }
            else if (board[i][j] === '#') {
                board[i][j] = 'O'
            }
        }
    }

    function dfs(i, j) {
        stack.push([i, j])
        board[i][j] = '#'
        while(stack.length) {
            const x = last(stack)[0],
                y = last(stack)[1]
            // 判断
            if(x-1 >= 0 && board[x-1][y] === 'O') {
                stack.push([x-1, y])
                board[x-1][y] = '#'
            }
            else if(x+1 < height && board[x+1][y] === 'O') {
                stack.push([x+1, y])
                board[x+1][y] = '#'
            }
            else if(y-1 >= 0 && board[x][y-1] === 'O') {
                stack.push([x, y-1])
                board[x][y-1] = '#'
            }
            else if(y+1 < width && board[x][y+1] === 'O') {
                stack.push([x, y+1])
                board[x][y+1] = '#'
            }
            else{
                stack.pop()
            }
        }
    }
    function last(arr){
        return arr[arr.length - 1]
    }

};
