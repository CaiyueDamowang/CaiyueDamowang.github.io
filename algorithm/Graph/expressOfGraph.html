function Graph(v) {
  this.vertices = v
  this.edges = 0
  this.adj = []

  for(let i = 0; i < this.vertices.length; i++) {     // 初始化边关系
    this.adj[i] = []
    this.adj[i].push("")
  }
}

Graph.prototype = {         // 无向图
  addEdge(v, w) {
    this.adj[v].push(w)
    this.adj[w].push(v)
    this.edges ++
  }
  showGraph() {
    for(let i = 0; i < this.vertices.length; i++>) {
      log(i + "->")
      for(let j = 0; j < this.adj[i].length; j++) {
        log(" " + this.adj[i][j] + " ")
      }
    }
  }
}
