# 原地单链表的反转

  • 思路: 原地反转的关键在于保证前一个和下一个结点的引用不丢失

    设置三个指针： pre 保存上一个节点 temp 保存下一个节点 node 指向当前节点

const reverse = (node)=>{
    if(!node)return null

    let pre = null
    let temp
    while (temp = node.next) {
        node.next = pre           // 反转指针
        pre = node
        node = temp               // 移动node指针
    }
    node.next = pre

    return node
}

const reverse2 = (node,temp)=>{

        temp = node.next
        node.next = pre

# 区间反转

  • 思路: 1.新建一个守卫，指向第一个结点（为了解决边界问题，防治出现区间前一个结点(front)就是区间开始结点(enter) 的情况) 2.将节点搜索到区间前一个结点 (front),区间开始结点(enter) 3.新建指针cur、pre、temp 反转区间的结点 4.退出循环后，cur指针指向的应该是区间后的一个结点 5.将front指向区间反转后的头结点pre, 6.将enter指向cur

如果没有想到建立一个守卫，很容易因为边界条件头晕

const reversBetween = (node, m, n) => {
    const originNode = new nodeList()
    originNode.next = node   // 新建一个虚拟结点是为解决边界问题

    const count = n - m
    let p = originNode
    for (let i = 0; i < m - 1; i++) {
        p = p.next
    } 

    let front = p // 区间前一个节点
    let cur = p.next,
        enter = p.next,   // 区间开始结点
        pre = null

    for (let j = 0; j <= count; j++) {    // 反转区间中的结点
        let temp = cur.next
        cur.next = pre
        pre = cur
        cur = temp
    }
    // 退出循环后， pre指向的是区间最后一个结点， cur指向的是区间后面的一个节点

    front.next = pre
    enter.next = cur
    return originNode.next
}
