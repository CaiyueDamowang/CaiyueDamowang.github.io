# 有效的括号字符串

给定一个只包含三种字符的字符串：（ ，）  和 *，写一个函数来检验这个字符串是否为有效字符串。有效字符串具有如下规则：

任何左括号 (  必须有相应的右括号 )。 任何右括号 )  必须有相应的左括号 ( 。 左括号 ( 必须在对应的右括号之前 )。 *  可以被视为单个右括号 ) ，或单个左括号 ( ，或一个空字符串。 一个空字符串也被视为有效字符串。

力扣

# 暴力解法：

每次碰到一个* 就生成两个新的字符串

  • 执行用时： 3336 ms
    lc 击败 7%
  • 内存耗时： 45.9 MB lc 击败 6%
var checkValidString = function(s) {
  let countLeft = 0;
  let countRight = 0;
  let countChar = 0;

  // 简单优化
  for (let i = 0; i < s.length; i++) {
    s[i] === "(" && countLeft++;
    s[i] === ")" && countRight++;
    s[i] === "*" && countChar++;
  }
  if (countChar < Math.abs(countLeft - countRight)) {
    return false;
  }

  // 开始遍历和判断
  const tokenList = [")", ""];
  let isValid = false;

  let arr = [s];
  while (arr.length) {
    const value = arr.pop();
    isValid = judge(value);
    if (isValid) return isValid;
  }

  return isValid;

  function judge(string) {
    const stack = [];
    for (let i = 0; i < string.length; i++) {
      if (stack.length > string.length - i) return false;
      if (string[i] === "(") {
        stack.push(string[i]);
      } else if (string[i] === ")") {
        if (!stack.length) return false;
        stack.pop();
      } else if (string[i] === "*") {
        string = string.slice(0, i) + "(" + string.slice(i + 1);

        const newStrArr = tokenList.map((token) => {
          return string.slice(0, i) + token + string.slice(i + 1);
        });
        arr = arr.concat(newStrArr);

        i--;
      }
    }
    if (stack.length) return false;
    return true;
  }
};

# 双向遍历

维护一个保存*的栈

维护一个保存括号的栈，每次碰到右括号，就在栈中取出一个左括号，如果没有左括号的话

就在通配符栈中提取一个*（此时可以保证通配符是在该右括号左边，并且左括号不足）

遍历结束后，情况如下

  • 通配符可能分配在左侧序列的中间或者右边
    • 此时需要判断左括号的数量是否小于通配符的数量，否则失败
  • 通配符可能都在左括号的左侧
    • 此时剩余的左括号都无法匹配，失败

      *****(((())

此情况代表了任意左右括号之间插入了任意多个通配符

也代表任意位置插入了任意多个可匹配的括号

此时反转遍历（从后向前）

通过右括号来消除左括号后， 检查是否有多余的左括号

  • 执行用时： 84 ms
    lc 击败 59%
  • 内存耗时： 38.3 MB lc 击败 6%
function checkValidString(s) {
  // 反转字符
  const reverseStr = s
    .split("")
    .reverse()
    .map((str) => {
        switch(str) {
            case '*': return str
            case '(': return ')'
            case ')': return '('
        }
    })
    .join("");

  return judge(s) && judge(reverseStr);

  function judge(string) {
    const charStack = [];
    const stack = [];

    for (let i = 0; i < string.length; i++) {
      switch (string[i]) {
        case "(":
          stack.push("(");
        case ")": {
          if (stack.length) {
            stack.pop();
          } else if (charStack.length) {
            charStack.pop();
          } else {
            return false;
          }
        }
        case "*": {
          charStack.push("*");
        }
      }
    }
    return Boolean(stack.length < charStack.length)

  }
}
